# Setup domain and calculate f and g
domain = np.linspace(0, 2*np.pi, 1000)
f = domain
g = np.sin(domain)

Compute the circular convolution $f * g$, and the Hadamard product $f \odot g$

In [ ]:
# Compute convolution using ifft and equation 8.52
f_g_conv = 1000 * ifft(fft(f) * fft(g))

# Compute hadamard product
f_g_prod = f * g
